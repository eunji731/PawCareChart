-- 5-1. 일정 관리 (다가오는 케어 및 예약 예약시스템)
CREATE TABLE schedules (
    id BIGSERIAL PRIMARY KEY,
    dog_id BIGINT NOT NULL REFERENCES dogs(id) ON DELETE CASCADE,
    title VARCHAR(200) NOT NULL,            -- 일정 제목 (예: 심장사상충 예방약 복용 등)
    schedule_date DATE NOT NULL,            -- 예정 날짜
    schedule_type VARCHAR(50),              -- 타입 (MEDICATION, VET_VISIT, GROOMING)
    is_completed BOOLEAN DEFAULT FALSE,     -- 완료 여부 (체크박스 체크 상태)
    memo TEXT,                              -- 메모/상세 내용
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 5-2. 기존 병원 기록 테이블(vet_records)에 복약 트래커 컬럼 추가
ALTER TABLE vet_records 
ADD COLUMN medication_start_date DATE,                  -- 약 복용 시작일
ADD COLUMN medication_days INTEGER,                     -- 총 복용 기간 (일수)
ADD COLUMN is_medication_completed BOOLEAN DEFAULT FALSE; -- 복약(과정) 조기/강제 종료 처리용

-- 5-3. 빠른 관찰 메모 (가벼운 건강 이슈 스냅샷)
CREATE TABLE health_logs (
    id BIGSERIAL PRIMARY KEY,
    dog_id BIGINT NOT NULL REFERENCES dogs(id) ON DELETE CASCADE,
    log_date DATE NOT NULL DEFAULT CURRENT_DATE, -- 관찰 기록일
    content TEXT NOT NULL,                  -- 메모 내용 (예: 노란 토 1회, 무른 변)
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 5-4. 증상 정규화 태그 테이블 (1위 피부염 등 통계 집계용)
CREATE TABLE symptom_tags (
    id BIGSERIAL PRIMARY KEY,
    dog_id BIGINT NOT NULL REFERENCES dogs(id) ON DELETE CASCADE,
    vet_record_id BIGINT REFERENCES vet_records(id) ON DELETE SET NULL, -- 어떤 병원 기록에서 시작되었는지
    symptom_name VARCHAR(100) NOT NULL,     -- 증상명 (예: '피부염', '외이염', '설사')
    severity VARCHAR(20),                   -- 심각도 (경증, 중증 등)
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);
